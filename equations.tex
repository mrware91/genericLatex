%%%%%%%%%%%%%%%%               Equation print functions
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand\myEquations[1]{\csname #1\endcsname}

\newcommand{\declareEquation}[2]{%
\expandafter\def\csname #1\endcsname{%
#2
}%
}

\newcommand{\splitEquation}[2]{
\begin{split}
\myEquations{#1}#2
\end{split}
}

\newcommand{\rePrintEquation}[3]{
\begin{equation}
\ifstrequal{#3}{split}{\splitEquation{#1}{#2}}{\myEquations{#1}#2}
\tag{\ref{#1} revisited}
\end{equation}
}

\newcommand{\printEquation}[3]{
\listcsgadd{eqList}{#1}
\begin{equation}\label{#1}
\ifstrequal{#3}{split}{\splitEquation{#1}{#2}}{\myEquations{#1}#2}
\end{equation}
}

\def\printAllEquations{%
\begin{description}
    \renewcommand*{\do}[1]{\item[##1:] \rePrintEquation{##1}{.}{split}}
    \dolistcsloop{eqList}
\end{description}%
}

\newcommand{\inlineEquation}[1]{%
\listcsgadd{eqList}{#1}%
$\myEquations{#1}$%
}
